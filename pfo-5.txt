#include <bits/stdc++.h>
using namespace std;

int main() {
    string T, C;
    cin >> T >> C;
    string password = "";
    for (int i = 0; i < C.size(); i++) {
        password += T[i];
        password += C[i];
    }
    if (T.size() > C.size()) {
        password += T.back();
    }

    cout << password;
}
